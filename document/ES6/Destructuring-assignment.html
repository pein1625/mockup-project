Destructuring Assignment

Array destructuring

Basic assignment

var foo = [1, 2, 3];
var [a, b, c] = foo;
console.log(
  a, // 1
  b, // 2
  c  // 3
);

Assignment separate from declaration

Ta có thể khai báo biến trước rồi sau đó mới gán

var a, b, c;
[a, b, c] = [1, 2, 3];

Default value

Có thể gán giá trị default cho biến,

var [a=1, b=2, c=3] = [10, undefined];
console.log(
  a, // 10
  b, // 2
  c  // 3
);

Swrapping variables

Ta có thể đổi giá trị giữa các biến một cách dễ dàng mà không cần tới biến tạm
ES6:

var a = 10;
var b = 20;
[b, a] = [a, b];
console.log(
  a, // 20
  b  // 10
);

ES5:
var a = 10;
var b = 20;
var tmp = a;
a = b;
b = tmp;
console.log(
  a, // 20
  b  // 10
);

Bỏ qua một số phần tử array

Nếu bạn chỉ muốn lấy một số phần tử của array

var foo = [1, 2, 3, 4];
var [, a, , b] = foo;
console.log(
  a, // 2
  b  // 4
);

Gán cho rest array

var foo = [1, 2, 3, 4];
var [a, ...b] = foo;
console.log(
  a, // 1
  b  // [2, 3, 4]
);

Lưu ý: rest variable luôn phải đứng cuối cùng của chuỗi.

Object desctructuring

Basic assignment

var o = { a: 1, b: 'hai' }
var { a, b } = o;
console.log(
  a, // 1
  b, // 'hai'
);

Ở đây biến a, b phải trùng tên với thuộc tính a, b trong object o

Assignment separate declaration

var o = { a: 1, b: 'hai' };
var a, b;
({ a, b } = o);
conosole.log(
  a, // 1
  b  // 'hai'
);

Lưu ý: Biểu thức ({ a, b } = 0) phải được đặc trong cặp dấu ngoặc đơn

Gán cho một tên mới

var o = { a: 1, b: 'hai' };
var { a: x, b: y } = o;
console.log( 
  a, // error: a chưa được định nghĩa
  b, // error: b chưa được định nghĩa
  x, // 1
  y, // 'hai'
);

Gán giá trị mặc định

Phần này cũng tương tự như array

var { a=1, b=2, c=3 } = { a: 10, b: undefined };
console.log(
  a, // 10
  b, // 2
  c  // 3
);

Gán cho một tên biến mới và thêm giá trị mặc định

var { a: x=10, b: y=20, c: z=30 } = { a: 1, b: undefined };
console.log(
  x, // 1
  y, // 20
  z, // 30
);

Đặt giá trị mặc định cho tham số của hàm (function)

ES6:

function foo({ name='john', age=30 } = {}){
  console.log(
    name,
    age
  );
}
var o = {
  name: 'mike',
  age: 11
}
foo(o); // mike 11
foo(); // john 3

ES5:
function foo(obj){
  var name, age;
  if (obj === undefined) obj = {};
  if (obj.name === undefined) name = 'john'
  else name = obj.name
  if (obj.age === undefined) age = 30 
  else age = obj.age
  console.log( 
    name,
    age
  );
}
var o = {
  name: 'mike',
  age: 11
}
foo(o); // mike 11 
foo(); // john 3

Nested object and array destructuring

var o = {
  a: 10,
  b: [1, {name: 'john', age: 30}]
};
var { a, b: [x, { name, age }]} = o;
console.log(
  a, // 10
  x, // 1
  name, // john
  age // 30
);

Destructuring và vòng lặp for..of

var o = [
  {
    name: 'john',
    age: 30
  },
  {
    name: 'mike'
  },
];
for ( var {name, age=10} of o) {
  console.log(name, age); // 'john' 30, 'mike' 10
}

Tách các trường từ object sau đó gán vào function parameter

function foo({name: { lastname }, age}){
  console.log(lastname, age);  
}
var o = {
  age: 30,
  name: {
    firstname: 'john',
    lastname: 'doe'
  }
}
foo(o); // 'doe' 30

Computed object property names and destructuring

let key = 'z';
let { [key]: foo } = { z: 'hello' };
console.log(foo); // 'hello'

Rest in Object Destructuring

var o = { a: 10, b: 20, c: 30, d: 40 };
var { b, ...o1} = o;
console.log(
  b, // 20
  o1, // { a: 10, c: 30, d: 40 }
);
